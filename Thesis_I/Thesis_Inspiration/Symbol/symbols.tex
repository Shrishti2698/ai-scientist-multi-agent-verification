\FrontMatterChapter{List of Symbols and Notations}

\begin{description}[style=nextline,leftmargin=1.7in,labelwidth=1.35in,font=\normalfont\bfseries]
\item[\(N\)] Number of evaluated benchmark instances.
\item[\(k\)] Rank cutoff used in Recall@\(k\) and Pass@\(k\) metrics.
\item[\(x_i\)] Feature vector for instance \(i\) before decision-gate prediction.
\item[\(p_{i,\mathrm{cal}}\)] Calibrated probability that PatchOps should act on instance \(i\).
\item[\(\tau\)] Decision threshold used by the calibrated act-versus-hold gate.
\item[\(\mathrm{ECE}\)] Expected calibration error.
\item[\(\mathrm{AURC}\)] Area under the risk-coverage curve.
\end{description}
