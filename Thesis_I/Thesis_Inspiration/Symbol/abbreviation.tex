\FrontMatterChapter{List of Abbreviations}

\begin{description}[style=nextline,leftmargin=1.7in,labelwidth=1.35in,font=\normalfont\bfseries]
\item[AUC] Area Under the Receiver Operating Characteristic Curve
\item[APR] Automated Program Repair
\item[AURC] Area Under the Risk-Coverage Curve
\item[BM25] Best Matching 25 lexical ranking model
\item[CI] Continuous Integration
\item[CV] Cross-Validation
\item[ECE] Expected Calibration Error
\item[F1] Harmonic mean of precision and recall
\item[LOO] Leave-One-Out evaluation
\item[LLM] Large Language Model
\item[PR] Pull Request
\item[RCA] Root Cause Analysis
\item[RAG] Retrieval-Augmented Generation
\item[ReAct] Reasoning and Acting
\item[Recall@k] Fraction of cases localized within the top-\(k\) ranked files
\item[Pass@1] Fraction of benchmark instances solved by the first generated repair
\item[ROC] Receiver Operating Characteristic
\item[RQ] Research Question
\item[TF-IDF] Term Frequency-Inverse Document Frequency
\end{description}
